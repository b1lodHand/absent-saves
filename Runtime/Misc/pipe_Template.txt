namespace com.absence.savesystem
{
    internal static class DataPipe
    {
        public static void SendData(object newData)
        {
            GameSaveData.Current = (GameSaveData)newData;
        }

        public static object RetrieveData()
        {
            return GameSaveData.Current;
        }

        public static void ResetData()
        {
            GameSaveData.Reset();
        }
    }
}
